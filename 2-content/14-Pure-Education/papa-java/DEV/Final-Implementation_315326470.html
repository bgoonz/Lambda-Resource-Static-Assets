@MappedSuperclass
@EntityListeners(AuditingEntityListener.class)
public class BaseEntity extends ResourceSupport
{
    public static ZonedDateTime DATE_INFINITY = ZonedDateTime.parse("9999-12-31T00:00:00Z");

    @Column(name = "in_utc", nullable = false, updatable = false)
    @CreatedDate
    @CreationTimestamp
    @Temporal(TemporalType.TIMESTAMP)
    private Date inUtc;

    @Column(name = "out_utc")
    private ZonedDateTime outUtc = DATE_INFINITY;

    @CreatedBy
    @JsonIgnore
    @Column(name = "created_by")
    private String createdBy;

    @LastModifiedBy
    @JsonIgnore
    @Column(name = "last_modified_by")
    private String lastModifiedBy;


    public Date getInUtc()
    {
        return inUtc;
    }

    public void setInUtc(Date inUtc)
    {
        this.inUtc = inUtc;
    }

    public ZonedDateTime getOutUtc()
    {
        return outUtc;
    }

    public void setOutUtc(ZonedDateTime outUtc)
    {
        this.outUtc = outUtc;
    }

    public String getCreatedBy()
    {
        return createdBy;
    }

    public void setCreatedBy(String createdBy)
    {
        this.createdBy = createdBy;
    }

    public String getLastModifiedBy()
    {
        return lastModifiedBy;
    }

    public void setLastModifiedBy(String lastModifiedBy)
    {
        this.lastModifiedBy = lastModifiedBy;
    }
}

@ControllerAdvice
@RestController
public class CustomizedResponseEntityExceptionHandler extends ResponseEntityExceptionHandler
{

    @ExceptionHandler(Exception.class)
    public final ResponseEntity<Object> handleAllExceptions(Exception ex, WebRequest request)
    {
        ExceptionResponse exceptionResponse = new ExceptionResponse(ZonedDateTime.now(), ex.getMessage(), request.getDescription(false));
        return new ResponseEntity(exceptionResponse, HttpStatus.INTERNAL_SERVER_ERROR);
    }


    @ExceptionHandler(NotFoundException.class)
    public final ResponseEntity<Object> handleReaderNotFoundException(NotFoundException ex, WebRequest request)
    {
        ExceptionResponse exceptionResponse = new ExceptionResponse(ZonedDateTime.now(), ex.getMessage(),
                request.getDescription(false));
        return new ResponseEntity(exceptionResponse, HttpStatus.NOT_FOUND);
    }


    @Override
    protected ResponseEntity<Object> handleMethodArgumentNotValid(MethodArgumentNotValidException ex,
                                                                  HttpHeaders headers, HttpStatus status, WebRequest request)
    {
        ExceptionResponse exceptionResponse = new ExceptionResponse(ZonedDateTime.now(), "Validation Failed",
                ex.getBindingResult().toString());
        return new ResponseEntity(exceptionResponse, HttpStatus.BAD_REQUEST);
    }
}

@Configuration
@Profile({"db"})
public class DataSourceConfig
{
    @Value("${datasource.driver-class-name}")
    String dataSourceDriverClass;

    @Value("${datasource.url}")
    String dataSourceUrl;

    @Value("${datasource.system.account.credref}")
    String credRef;

    @Bean
    public DataSource getDataSource()
    {
        CredentialLookUp<SystemAccount> credentialLookUp = new CredentialLookUp<>();
        SystemAccount systemAccount = credentialLookUp.getCredentialObject(credRef);
        String systemAccountPassword = systemAccount.getPassword();
        String systemAccountName = systemAccount.getAccountName();
        String[] systemAccountParts = systemAccountName.split("\\\\");
        return DataSourceBuilder
                .create()
                .username(systemAccountParts[1])
                .password(systemAccountPassword)
                .url(dataSourceUrl)
                .driverClassName(dataSourceDriverClass)
                .build();
    }
}

@RestController
@RepositoryRestController
public class EmployeeController
{
    private Logger logger = LoggerFactory.getLogger(getClass());

    @Autowired
    private EmployeeRepository employeeRepository;
    @Autowired
    private TeamRepository teamRepository;


    @GetMapping("/employees")
    public ResponseEntity<Resources<Resource<EmployeePresentation>>> getEmployees()
    {
        List<Resource<EmployeePresentation>> resourceResult = new ArrayList<>();

        List<Employee> employees = employeeRepository.findAll();
        for(Employee employee : employees)
        {
            EmployeePresentation employeePresentation = new EmployeePresentation(employee);
            Resource<EmployeePresentation> employeePresentationResource = new Resource<>(employeePresentation, createEmployeeSelfLink(employee.getID()));
            employeePresentationResource.add(createTeamLink(employee.getTeam()));

            resourceResult.add(employeePresentationResource);
        }

        Resources<Resource<EmployeePresentation>> resources = new Resources<>(resourceResult);
        resources.add(linkTo(methodOn(getClass()).getEmployees()).withSelfRel());
        return ResponseEntity.ok(resources);
    }

    @GetMapping("/employees/{id}")
    public ResponseEntity<Resource<EmployeePresentation>> getEmployees(@PathVariable("id") int id)
    {
        Optional<Employee> employeeOptional = employeeRepository.findById(id);
        if(!employeeOptional.isPresent())
        {
            throw new NotFoundException("employee-" + id);
        }

        Employee employee = employeeOptional.get();
        EmployeePresentation employeePresentation = new EmployeePresentation(employee);
        Resource<EmployeePresentation> employeePresentationResource = getEmployeePresentationResource(employeePresentation);
        employeePresentationResource.add(createTeamLink(employee.getTeam()));

        return ResponseEntity.ok(employeePresentationResource);
    }

    @PostMapping("/employees")
    public ResponseEntity<?> createEmployee(@RequestBody EmployeePresentation employeePresentation)
    {
        Employee employee = createEmployeeInternal(employeePresentation);

        URI location = ServletUriComponentsBuilder
                .fromCurrentRequest()
                .path("/{id}")
                .buildAndExpand(employee.getID()).toUri();

        return ResponseEntity.created(location).build();
    }

    //  note internal visibility
    Employee createEmployeeInternal(EmployeePresentation employeePresentation)
    {
        Optional<Team> teamOptional = teamRepository.findById(employeePresentation.getTeamId());
        if(!teamOptional.isPresent())
        {
            throw new NotFoundException("team-" + employeePresentation.getTeamId());
        }

        Team team = teamOptional.get();

        Employee employee = new Employee(employeePresentation);
        employee.setTeam(team);

        employeeRepository.save(employee);

        return employee;
    }


    @PutMapping("/duties/{id}")
    public ResponseEntity<?> updateEmployee(@PathVariable("id") int id, @RequestBody EmployeePresentation employeePresentation)
    {
        Optional<Employee> employeeOptional = employeeRepository.findById(id);
        if(!employeeOptional.isPresent())
        {
            throw new NotFoundException("employee-" + id);
        }

        Employee existingEmployee = employeeOptional.get();
        //  Create data for audit record
        Employee auditEmployee = new Employee(existingEmployee);

        //  Modify existing record with new values
        if(StringUtils.hasText(employeePresentation.getName()))
        {
            existingEmployee.setName(employeePresentation.getName());
        }
        if(StringUtils.hasText(employeePresentation.getGuid()))
        {
            existingEmployee.setGuid(employeePresentation.getGuid());
        }
        if(StringUtils.hasText(employeePresentation.getKerberos()))
        {
            existingEmployee.setKerberos(employeePresentation.getKerberos());
        }
        if(StringUtils.hasText(employeePresentation.getEmployeeNumber()))
        {
            existingEmployee.setEmployeeNumber(employeePresentation.getEmployeeNumber());
        }
        if(StringUtils.hasText(employeePresentation.getTitle()))
        {
            existingEmployee.setTitle(employeePresentation.getTitle());
        }
        if(StringUtils.hasText(employeePresentation.getPhone()))
        {
            existingEmployee.setPhone(employeePresentation.getPhone());
        }
        if(StringUtils.hasText(employeePresentation.getLocation()))
        {
            existingEmployee.setLocation(employeePresentation.getLocation());
        }
        if(StringUtils.hasText(employeePresentation.getDepartment()))
        {
            existingEmployee.setDepartment(employeePresentation.getDepartment());
        }

        employeeRepository.save(existingEmployee);
        //  Create Audit record
        employeeRepository.save(auditEmployee);

        URI location = ServletUriComponentsBuilder
                .fromCurrentRequest()
                .buildAndExpand(existingEmployee).toUri();

        return ResponseEntity.created(location).build();
    }

    private Resource<EmployeePresentation> getEmployeePresentationResource(EmployeePresentation employeePresentation)
    {
        Resource<EmployeePresentation> resource = new Resource<>(employeePresentation);
        ControllerLinkBuilder link = linkTo(methodOn(getClass()).getEmployees());
        resource.add(link.withRel("all"));
        return resource;
    }

    private Link createEmployeeSelfLink(int employeeId)
    {
        return linkTo(methodOn(getClass()).getEmployees(employeeId)).withRel("self");
    }

    private Link createTeamLink(Team team)
    {
        return linkTo(methodOn(TeamController.class).getTeams(team.getID())).withRel("team");
    }

}

    private EmployeeRepository employeeRepository;

    @GetMapping("/teams")
    public ResponseEntity<Resources<Resource<TeamPresentation>>> getTeams()
    {
        List<Resource<TeamPresentation>> resourceResult = new ArrayList<>();

        List<Team> teams = teamRepository.findAll();
        for(Team team : teams)
        {
            TeamPresentation teamPresentation = getTeamPresentationWithEmployees(team);
            Resource<TeamPresentation> teamPresentationResource = new Resource<>(teamPresentation, createTeamSelfLink(team.getID()));

            resourceResult.add(teamPresentationResource);
        }

        Resources<Resource<TeamPresentation>> resources = new Resources<>(resourceResult);
        resources.add(linkTo(methodOn(getClass()).getTeams()).withSelfRel());
        return ResponseEntity.ok(resources);
    }

    @GetMapping("/teams/{id}")
    public ResponseEntity<Resource<TeamPresentation>> getTeams(@PathVariable("id") int id)
    {
        Optional<Team> teamOptional = teamRepository.findById(id);
        if(!teamOptional.isPresent())
        {
            throw new NotFoundException("team-" + id);
        }

        Team team = teamOptional.get();
        TeamPresentation teamPresentation = getTeamPresentationWithEmployees(team);
        Resource<TeamPresentation> teamPresentationResource = getTeamPresentationResource(teamPresentation);

        return ResponseEntity.ok(teamPresentationResource);
    }


    @PostMapping("/teams")
    public ResponseEntity<?> createTeam()
    {
        Team team = createTeamInternal();

        URI location = ServletUriComponentsBuilder
                .fromCurrentRequest()
                .path("/{id}")
                .buildAndExpand(team.getID()).toUri();

        return ResponseEntity.created(location).build();
    }

    //  Note internal visiblity
    Team createTeamInternal()
    {
        Team team = new Team();
        teamRepository.save(team);
        return team;
    }

    //  Note internal visibility
    TeamPresentation getTeamPresentationWithEmployees(Team team)
    {
        TeamPresentation teamPresentation = getTeamPresentation(team);
        teamPresentation.setEmployeePresentations(getTeamEmployeePresentations(team.getID()));
        return teamPresentation;
    }

    private List<EmployeePresentation> getTeamEmployeePresentations(int teamId)
    {
        List<EmployeePresentation> employeePresentations = new ArrayList<>();

        List<Employee> employees = employeeRepository.findByTeamId(teamId);
        for(Employee employee : employees)
        {
            employeePresentations.add(new EmployeePresentation(employee));
        }

        return employeePresentations;
    }

    private TeamPresentation getTeamPresentation(Team team)
    {
        TeamPresentation teamPresentation = new TeamPresentation(team);
        teamPresentation.add(linkTo(methodOn(getClass()).getTeams(team.getID())).withSelfRel());
        return teamPresentation;
    }

    private Resource<TeamPresentation> getTeamPresentationResource(TeamPresentation teamPresentation)
    {
        Resource<TeamPresentation> resource = new Resource<>(teamPresentation);
        ControllerLinkBuilder link = linkTo(methodOn(getClass()).getTeams());
        resource.add(link.withRel("all"));
        return resource;
    }


    private Link createTeamSelfLink(int teamId)
    {
        return linkTo(methodOn(getClass()).getTeams(teamId)).withRel("self");
    }

@RestController
@RepositoryRestController
public class RmsNodeController
{
    private Logger logger = LoggerFactory.getLogger(getClass());

    @Autowired
    private RmsNodeRepository rmsNodeRepository;
    @Autowired
    private RmsNodeMapRepository rmsNodeMapRepository;
    @Autowired
    private EmployeeController employeeController;
    @Autowired
    private TeamController teamController;


    @GetMapping("/rmsnodes/code/{code}")
    public ResponseEntity<Resource<RmsNodePresentation>> getRmsNodesByCode(@PathVariable("code") String code)
    {
        Optional<RmsNode> rmsNodeOptional = rmsNodeRepository.findByCode(code);
        if(!rmsNodeOptional.isPresent())
        {
            throw new NotFoundException("RmsNode-" + code);
        }

        return getRmsNodes(rmsNodeOptional.get().getID());
    }

    @GetMapping("/rmsnodes/{id}")
    public ResponseEntity<Resource<RmsNodePresentation>> getRmsNodes(@PathVariable("id") int id)
    {
        Optional<RmsNode> rmsNodeOptional = rmsNodeRepository.findById(id);
        if(!rmsNodeOptional.isPresent())
        {
            throw new NotFoundException("RmsNode-" + id);
        }

        RmsNode rmsNode = rmsNodeOptional.get();

        RmsNodePresentation rmsNodePresentation = new RmsNodePresentation(rmsNode);
        getRmsNodePresentationWithChildren(rmsNode, rmsNodePresentation);

        Resource<RmsNodePresentation> rmsNodePresentationResource = new Resource<>(rmsNodePresentation);
        rmsNodePresentationResource.add(createRmsNodeSelfLink(id));

        return ResponseEntity.ok(rmsNodePresentationResource);
    }

    private void saveRmsNodes(RmsNodeInput rmsNodeInput, RmsNode rmsNodeParent)
    {
        rmsNodeParent.setName(rmsNodeInput.getName());
        rmsNodeParent.setNodeType(rmsNodeInput.getNodeType());
        rmsNodeParent.setCode(rmsNodeInput.getCode());

        List<Employee> employees = rmsNodeInput.getTeam();
        if(!employees.isEmpty())
        {
            Team team = teamController.createTeamInternal();

            for(Employee employee : employees)
            {
                employee.setTeam(team);
                employeeController.createEmployeeInternal(new EmployeePresentation(employee));
            }
            rmsNodeParent.setTeam(team);
        }

        for(RmsNodeInput childRmsNodeInput : rmsNodeInput.getChildren())
        {
            RmsNode rmsNodeChild = new RmsNode();
            saveRmsNodes(childRmsNodeInput, rmsNodeChild);

            rmsNodeMapRepository.save(new RmsNodePair(rmsNodeParent, rmsNodeChild));
        }
    }

    @PostMapping("/rmsnodes")
    public ResponseEntity<?> createRmsNodes(@RequestBody RmsNodeInput rmsNodeInput)
    {
        RmsNode rmsNode = new RmsNode();
        saveRmsNodes(rmsNodeInput, rmsNode);

        URI location = ServletUriComponentsBuilder
                .fromCurrentRequest()
                .path("/{id}")
                .buildAndExpand(rmsNode.getID()).toUri();

        return ResponseEntity.created(location).build();
    }

    private void getRmsNodePresentationWithChildren(RmsNode rmsNode, RmsNodePresentation rmsNodePresentation)
    {
        Team team = rmsNode.getTeam();
        if(team != null)
        {
            rmsNodePresentation.setTeamPresentation(teamController.getTeamPresentationWithEmployees(team));
        }

        List<RmsNodePair> rmsNodePairs = rmsNodeMapRepository.findByParentId(rmsNode.getID());
        List<RmsNode> childrenRmsNodes = rmsNodePairs.stream().map(RmsNodePair::getChildRmsNode).collect(Collectors.toList());
        for(RmsNode childRmsNode : childrenRmsNodes)
        {
            RmsNodePresentation rmsChildNodePresentation = new RmsNodePresentation(childRmsNode);
            rmsChildNodePresentation.add(createRmsNodeSelfLink(childRmsNode.getID()));
            getRmsNodePresentationWithChildren(childRmsNode, rmsChildNodePresentation);
            rmsNodePresentation.addChildPresenation(rmsChildNodePresentation);
        }

        if(!rmsNodePresentation.getChildPresentations().isEmpty())
        {
            rmsNodePresentation.add(linkTo(methodOn(getClass()).getRmsNodes(rmsNode.getID())).withSelfRel());
        }
    }

    private Link createRmsNodeSelfLink(int rmsNodeId)
    {
        return linkTo(methodOn(getClass()).getRmsNodes(rmsNodeId)).withRel("self");
    }

}
