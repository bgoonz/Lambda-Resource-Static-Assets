public class Employee

    @GeneratedValue(strategy = GenerationType.SEQUENCE)
    @Column(name="ID", updatable = false, nullable = false)
    private int ID;
    private String employeeCode;
    private String employeeName;

    public int getID()
    {
        return ID;
    }

    {
        this.ID = ID;

    public String getEmployeeCode()

        return employeeCode;
    }

    public void setEmployeeCode(String employeeCode)
    {
        this.employeeCode = employeeCode;
    }

    public String getEmployeeName()
    {
        return employeeName;
    }

    public void setEmployeeName(String employeeName)
    {
        this.employeeName = employeeName;
    }
}

@Entity
@Table(name = "PROJECT_EMPLOYEE_MAP")
public class ProjectEmployeePair
{
    @Id
    @GeneratedValue(strategy = GenerationType.SEQUENCE)
    @Column(name="ID", updatable = false, nullable = false)
    private int ID;
    @ManyToOne
    @JoinColumn(name = "project_id")
    private Project project;
    @ManyToOne
    @JoinColumn(name = "employee_id")
    private Employee employee;

    public int getID()
    {
        return ID;
    }

    public void setID(int ID)
    {
        this.ID = ID;
    }

    public Project getProject()
    {
        return project;
    }

    public void setProject(Project project)
    {
        this.project = project;
    }

    public Employee getEmployee()
    {
        return employee;
    }

    public void setEmployee(Employee employee)
    {
        this.employee = employee;
    }
}


@Entity
@Table(name = "EMPLOYEE_SUBSCRIPTION_MAP")
public class EmployeeSubscriptionPair
{
    @Id
    @GeneratedValue(strategy = GenerationType.SEQUENCE)
    @Column(name="ID", updatable = false, nullable = false)
    private int ID;
    @ManyToOne
    @JoinColumn(name = "employee_id")
    private Employee employee;

    @ManyToOne
    @JoinColumn(name = "subscription_id")
    private Subscription subscription;

    public int getID()
    {
        return ID;
    }

    public void setID(int ID)
    {
        this.ID = ID;
    }

    public Employee getEmployee()
    {
        return employee;
    }

    public void setEmployee(Employee employee)
    {
        this.employee = employee;
    }

    public Subscription getSubscription()
    {
        return subscription;
    }

    public void setSubscription(Subscription subscription)
    {
        this.subscription = subscription;
    }
}

@RestController
@RepositoryRestController
public class ProjectController
{
    @Autowired
    private ProjectRepository projectRepository;

    @GetMapping("/projects")
    public ResponseEntity<List<Project>> getProjects()
    {
        List<Project> projects = projectRepository.findAll();
        return ResponseEntity.ok(projects);
    }

    @GetMapping("/projects/{id}")
    public ResponseEntity<Project> getProjects(@PathVariable("id") int id)
    {
        Optional<Project> projectOptional = projectRepository.findById(id);
        return ResponseEntity.ok(projectOptional.get());
    }
}

@RestController
@RepositoryRestController
public class ProjectEmployeeMapController
{
    @Autowired
    private ProjectEmployeeMapRepository projectEmployeeMapRepository;

    @GetMapping("/projects/employees")
    public ResponseEntity<List<Project>> getProjectsEmployees()
    {
        List<ProjectEmployeePair> projectEmployeePairs = projectEmployeeMapRepository.findAll();
        Map<Project, List<Employee>> groupedMap = projectEmployeePairs.stream()
                .collect(Collectors
                        .groupingBy(ProjectEmployeePair::getProject,
                                Collectors.mapping(ProjectEmployeePair::getEmployee, Collectors.toList())));

        List<Project> result = new ArrayList<>();
        for(Map.Entry<Project,List<Employee>> e : groupedMap.entrySet())
        {
            result.add(e.getKey());
        }

        return ResponseEntity.ok(result);
    }

    @GetMapping("/employees/projects")
    public ResponseEntity<List<Employee>> getEmployeesProjects()
    {
        List<ProjectEmployeePair> projectEmployeePairs = projectEmployeeMapRepository.findAll();
        Map<Employee, List<Project>> groupedMap = projectEmployeePairs.stream()
                .collect(Collectors
                        .groupingBy(ProjectEmployeePair::getEmployee,
                                Collectors.mapping(ProjectEmployeePair::getProject, Collectors.toList())));

        List<Employee> result = new ArrayList<>();
        for(Map.Entry<Employee,List<Project>> e : groupedMap.entrySet())
        {
            result.add(e.getKey());
        }

        return ResponseEntity.ok(result);
    }
}
